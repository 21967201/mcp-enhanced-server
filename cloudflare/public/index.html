Deploy Your Own

# Clone and install
git clone https://github.com/YOUR_USER/mcp-enhanced-server.git
cd mcp-enhanced-server/cloudflare
npm install

# Create D1 database
npx wrangler d1 create mcp-enhanced-tasks

# Update wrangler.toml with your database_id

# Run migrations
npm run db:migrate

# Deploy to Cloudflare Workers
npm run deploy
